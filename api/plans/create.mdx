```python Python
)

    'free_units': 10000,
    'package_size': 1000
  }
)

  billable_metric_id='id,
  charge_model='percentage',
  pay_in_advance=false,

  properties={
    'rate': '0.5',
    'fixed_amount': '1',
    'free_units_per_events': 3,
    'free_units_per_total_aggregation': None
  }
)

volume_charge = Charge(
  billable_metric_id='id,
  charge_model='volume',
  pay_in_advance=false,
  min_amount_cents=0,
  properties={
    'volume_ranges': [
      {
        'to_value': 10,
        'from_value': 0,

      {

    ]
  }
)

  standard_charge,

  package_charge,
  percentage_charge,
  volume_charge
])

plan = Plan(
  name='Startup',
  code='startup',
  amount_cents=10000,
  amount_currency='USD',
  description='This is a startup plan.',
  interval='monthly',
  pay_in_advance=True,
  charges=charges
)

try:
    client.plans.create(plan)
except LagoApiError as e:
    repair_broken_state(e)  # do something on error or raise your own exception
```

```go Go
import "fmt"
import "github.com/getlago/lago-go-client"

func main() {
lagoClient := lago.New().
    SetApiKey("__YOUR_API_KEY__")

planInput := &lago.PlanInput{
    Name:           "Startup",
    Code:           "startup",
    Interval:       lago.PlanMonthly,
    PayInAdvance:   true,
    AmountCents:    10000,
    AmountCurrency: lago.USD,
}

plan, err := lagoClient.Plan().Create(planInput)
if err != nil {
    // Error is *lago.Error
    panic(err)
}

// plan is *lago.Plan
fmt.Println(plan)
}
```
